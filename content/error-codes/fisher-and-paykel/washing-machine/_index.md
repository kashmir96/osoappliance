    const brand = errorCode.querySelector("strong").textContent;

    } else {
      errorCode.style.display = "none";
    }
  });
});

errorCodeClear.addEventListener("click", function() {
  errorCodeSearch.value = ";

    errorCode.style.display = "block";
  });
});
